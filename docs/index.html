Robot Car robot.py 
from ludwig import Robot, Motor

car = Robot()
left = Motor(pin_a=17, pin_b=27)
right = Motor(pin_a=22, pin_b=23)

car.drive(left, right, speed=80)
Smart Home home.py 
from ludwig import Home, Light, Sensor

home = Home()
light = Light(pin=18)
motion = Sensor.motion(pin=24)

if motion.detect():
    light.on()
Voice Assistant assistant.py 
from ludwig import Assistant

bot = Assistant()

@bot.on("lights on")
def lights_on():
    home.lights.on()

bot.listen()
Garden garden.py 
from ludwig import Garden, Sensor, Pin

garden = Garden()
moisture = Sensor.moisture(pin=34)
pump = Pin(25, output=True)

if moisture.read() < 30:
    pump.on()

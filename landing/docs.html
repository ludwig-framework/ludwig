Ludwig Documentation

Ludwig is a Python framework for building web APIs, IoT systems, and AI applications with one consistent, zero-dependency core.

Installation

terminal
# Core framework (zero dependencies)
$ pip install ludwig

# With AI features
$ pip install ludwig[ai]

# With computer vision
$ pip install ludwig[vision]

# With Raspberry Pi GPIO
$ pip install ludwig[pi]

# Everything
$ pip install ludwig[all]

Quick Start

Create a new project with a template, or start from scratch:

terminal
$ ludwig new my-api --template api
$ cd my-api
$ python app.py

Or write it yourself:

app.py
from ludwig import App

app = App(name="My App")

@app.get("/")
def home(req):
    return {"status": "ok"}

app.run(debug=True)

Web Server

Ludwig includes a built-in HTTP server with no external dependencies. Use the App class for unified Web+IoT+AI apps, or Web standalone.

app.py
from ludwig import App

app = App(name="Web Demo")

# Route parameters with :param syntax
@app.get("/users/:id")
def get_user(req):
    return {"id": req.params["id"]}

@app.post("/users")
def create_user(req):
    return req.json, 201

@app.put("/users/:id")
def update_user(req):
    return {"updated": req.params["id"]}

@app.delete("/users/:id")
def delete_user(req):
    return {"deleted": True}

app.run(debug=True)

Routes

Register routes with decorators. Available methods:

MethodExample
@app.get(path)@app.get("/items")
@app.post(path)@app.post("/items")
@app.put(path)@app.put("/items/:id")
@app.delete(path)@app.delete("/items/:id")
@app.route(path, methods)@app.route("/x", ["GET","POST"])
@app.api(path)Registers GET, POST, PUT, DELETE all at once

Request & Response

Route handlers receive a Request object with:

  • req.params — URL parameters (:id values)
  • req.query — Query string as dict
  • req.json — Parsed JSON body
  • req.headers — Request headers
  • req.method — HTTP method
  • req.path — Request path

Return a dict or list for JSON, a str for HTML, or a Response object for full control.

Database

Ludwig includes a lightweight ORM backed by SQLite (or PostgreSQL with ludwig[postgres]). Define models as dataclasses.

Model ORM

models.py
from ludwig import Database, Model
from dataclasses import dataclass

db = Database("app.db")

@dataclass
class User(Model):
    name: str
    email: str
    age: int = 0

# Create
user = User.create(name="Alice", email="a@b.com")

# Read
found = User.find(user.id)
all_users = User.all()
adults = User.where("age", ">=", 18)

# Update
found.name = "Alice Smith"
found.save()

# Delete
found.delete()

# Serialize
found.to_dict()   # {"id": 1, "name": "...", ...}
found.to_json()   # JSON string
Note: The table is auto-created on first query. Table name defaults to the lowercase class name + "s" (e.g. User → users).

Query Builder

query.py
# Fluent query builder
users = db.table("users").where("age", ">", 18).order_by("name").get()

# First result
user = db.table("users").where("email", "alice@test.com").first()

# Count
count = db.table("users").count()

# Raw SQL
rows = db.query("SELECT * FROM users WHERE name = ?", ["Alice"])

IoT Overview

Ludwig's IoT layer provides hardware-agnostic GPIO with automatic board detection and simulation fallback. Import everything from ludwig.iot.

Simulation mode: When no hardware is detected, all GPIO operations print [SIM] messages. Write and test your code on any desktop, then deploy to a Raspberry Pi without changing a line.

Pins & GPIO

pin.py
from ludwig.iot import Pin

# Output pin
led = Pin(17, mode="output")
led.on()
led.off()
led.toggle()

# Input pin with pull-up
button = Pin(18, mode="input", pull="up")
if button.read():
    print("Pressed!")

# PWM (dimming, speed control)
pwm = led.pwm()
pwm.duty(50)  # 50% duty cycle
pwm.stop()

Sensors

All sensors are created via factory methods on the Sensor class:

sensors.py
from ludwig.iot import Sensor

motion = Sensor.motion(pin=4)
temp   = Sensor.temperature(pin=7, type="dht22")
humid  = Sensor.humidity(pin=7)
dist   = Sensor.ultrasonic(trigger=23, echo=24)
soil   = Sensor.moisture(pin=0)
light  = Sensor.light(pin=1)
door   = Sensor.door(pin=5)
window = Sensor.window(pin=6)

# Read values
print(motion.read())    # True/False
print(temp.read())      # float (celsius)
print(dist.read())      # float (cm)

# Event handlers
@motion.on_detected
def on_motion():
    print("Movement!")

Actuators

actuators.py
from ludwig.iot import Light, Motor, Servo, Relay, Buzzer

# Light — on/off/dim/blink
light = Light(pin=17, dimmable=True)
light.on()
light.brightness(50)
light.blink(times=3)
light.off()

# Motor — H-bridge with speed
motor = Motor(forward=17, backward=18, enable=12)
motor.forward(speed=75)
motor.stop()

# Servo — 0 to 180 degrees
servo = Servo(pin=18)
servo.angle(90)
servo.center()

# Relay — high-power devices
pump = Relay(pin=17)
pump.on()

# Buzzer — tones and melodies
buzzer = Buzzer(pin=18)
buzzer.beep()
buzzer.tone(440, 0.5)

Robot

robot.py
from ludwig.iot import Robot, Sensor

# 2-wheel differential drive
robot = Robot.car(
    left={"forward": 17, "backward": 18},
    right={"forward": 22, "backward": 23},
)

# Add obstacle sensor
robot.add_sensor("front",
    Sensor.ultrasonic(trigger=24, echo=25))

# Movement
robot.forward(speed=50)
robot.backward(duration=0.5)
robot.turn(90)
robot.stop()

# Obstacle avoidance callback
@robot.on_obstacle
def avoid():
    robot.stop()
    robot.backward(duration=0.5)
    robot.turn(90)
    robot.forward()

# Start monitoring loop
robot.run()

Smart Home

home.py
from ludwig.iot import Home, Light, Sensor

home = Home()

# Define rooms
living = home.room("living_room")
living.add(Light(pin=17, name="ceiling", dimmable=True))
living.add(Sensor.motion(pin=4))

bedroom = home.room("bedroom")
bedroom.add(Light(pin=18, name="main"))

# Time-based automations
@home.at("sunset")
def evening():
    living.light("ceiling").on()
    living.light("ceiling").brightness(70)

@home.at("23:00")
def bedtime():
    home.all_lights_off()

# Sensor-triggered automations
@home.when(lambda: living.motion.read())
def motion_light():
    living.light("ceiling").on(duration=300)

home.run()

Alarm

alarm.py
from ludwig.iot import Alarm, Sensor

alarm = Alarm()
alarm.add(Sensor.motion(pin=4), zone="living_room")
alarm.add(Sensor.door(pin=5), zone="front_door")
alarm.siren(pin=17)

@alarm.on_triggered
def alert(zone, sensor):
    alarm.send_sms("+1234567890", f"Intruder in {zone}!")

alarm.arm()
alarm.run()

Garden

garden.py
from ludwig.iot import Garden

garden = Garden()
garden.add_plant("tomatoes",
    moisture_pin=0, pump_pin=17,
    threshold=30, water_duration=10)

garden.add_plant("herbs",
    moisture_pin=1, pump_pin=18,
    threshold=40)

@garden.check_every(hours=1)
def water_check(plant):
    if plant.needs_water:
        plant.water(seconds=10)
        garden.log(f"Watered {plant.name}")

garden.run()

AI Overview

Ludwig's AI modules provide voice assistants, computer vision, and natural language automation. Install with pip install ludwig[ai] or ludwig[vision].

Voice Assistant

assistant.py
from ludwig.ai import Assistant

assistant = Assistant(
    name="Ludwig",
    model="gpt-4o",
    voice="alloy",
)

# Chat (text only, no mic needed)
reply = assistant.chat("What's the weather?")

# Voice loop with wake word
@assistant.on_wake("Hey Ludwig")
def handle():
    cmd = assistant.listen()
    resp = assistant.think(cmd)
    assistant.speak(resp)

# Command routing
@assistant.on_command("turn on lights")
def lights_on():
    home.all_lights_on()

assistant.run()

Computer Vision

vision.py
from ludwig.ai import Vision

vision = Vision(camera=0, model="yolo", confidence=0.5)

# Per-label callbacks
@vision.on_detect("person")
def person(det):
    print(f"Person at {det.center} ({det.confidence:.0%})")

# Manual detection
frame = vision.capture()
dets = vision.detect(frame)

# Object tracking
vision.track("person")
vision.run(display=True)

Automator

Define automation rules in natural language:

automator.py
from ludwig.ai import Automator

auto = Automator(model="gpt-4o")
auto.context = {
    "lights": home.all_lights,
    "temperature": living.temperature,
}

auto.add_rule("Turn off all lights at midnight")
auto.add_rule("If temperature > 28, turn on AC")

auto.run()

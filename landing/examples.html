basic

Web

app.py
from ludwig import App

app = App(name="My App")

todo_api

Full CRUD REST API with SQLite database and the Model ORM.

Web
$ludwig new my-app --template todo_api
app.py
from dataclasses import dataclass
from ludwig import App, Database, Model

app = App(name="Todo API")
db = Database("todos.db")

@dataclass
class Todo(Model):
    title: str
    completed: bool = False

@app.get("/todos")
def list_todos(req):
    todos = Todo.all()
    return [t.to_dict() for t in todos]

@app.post("/todos")
def create_todo(req):
    todo = Todo.create(**req.json)
    return todo.to_dict()

@app.delete("/todos/:id")
def delete_todo(req):
    todo = Todo.find(int(req.params["id"]))
    if not todo:
        return {"error": "Not found"}, 404
    todo.delete()
    return {"deleted": True}

if __name__ == "__main__":
    app.run(debug=True)

robot

Obstacle-avoiding robot car with ultrasonic sensor and differential drive.

IoT
$ludwig new my-app --template robot
app.py
from ludwig.iot import Robot, Sensor

robot = Robot.car(
    left={"forward": 17, "backward": 18, "enable": 12},
    right={"forward": 22, "backward": 23, "enable": 13},
)

robot.add_sensor("front", Sensor.ultrasonic(trigger=24, echo=25))

@robot.on_obstacle
def avoid_obstacle():
    robot.stop()
    robot.backward(duration=0.5)
    robot.turn(90)
    robot.forward()

if __name__ == "__main__":
    robot.forward(speed=50)
    robot.run(obstacle_distance=20)

alarm

alarm = Alarm()

smart-home

Home automation with rooms, dimmable lights, motion sensors, and scheduled actions.

IoT
$ludwig new my-app --template smart-home
app.py
from ludwig.iot import Home, Light, Sensor

home = Home()

living = home.room("living_room")
living.add(Light(pin=17, name="ceiling", dimmable=True))
living.add(Sensor.motion(pin=4))

bedroom = home.room("bedroom")
bedroom.add(Light(pin=18, name="main"))

@home.at("sunset")
def evening_mode():
    living.light("ceiling").on()
    living.light("ceiling").brightness(70)

@home.at("23:00")
def bedtime():
    home.all_lights_off()

if __name__ == "__main__":
    home.run()

garden

Automated plant watering with moisture sensors and scheduled checks.

IoT
$ludwig new my-app --template garden
app.py
from ludwig.iot import Garden

garden = Garden()

garden.add_plant("tomatoes",
    moisture_pin=0, pump_pin=17, threshold=30)
garden.add_plant("herbs",
    moisture_pin=1, pump_pin=18, threshold=40)

@garden.check_every(hours=1)
def hourly_check(plant):
    if plant.needs_water:
        plant.water(seconds=10)
        garden.log(f"Watered {plant.name}")

if __name__ == "__main__":
    garden.run()

door-lock

Smart door lock with PIN entry, web UI, and relay-controlled locking.

IoT
$ludwig new my-app --template door-lock
app.py
from ludwig import App
from ludwig.iot import Relay

app = App()
lock = Relay(pin=11)
PIN = "1234"

@app.route("/status")
def status(req):
    return {"locked": not lock.is_on()}

@app.route("/unlock", methods=["POST"])
def unlock(req):
    if req.json.get("pin") == PIN:
        lock.on()
        return {"success": True, "message": "Unlocked"}
    return {"success": False, "message": "Invalid PIN"}

@app.route("/lock", methods=["POST"])
def do_lock(req):
    lock.off()
    return {"success": True}

if __name__ == "__main__":
    app.run(port=8000)

from ludwig.ai import Assistant

assistant = Assistant(
    name="Ludwig",
    model="gpt-4o",
    voice="alloy",
)

@assistant.on_wake("Hey Ludwig")
def handle_wake():
    command = assistant.listen()
    response = assistant.think(command)
    assistant.speak(response)

vision = Vision(camera=0, model="yolo")
